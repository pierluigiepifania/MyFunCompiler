fun opInt ( integer num1, integer num2, integer tipoop, out integer ri) : integer
integer result;
if tipoop = 1 then
result := num1 + num2;
end if;
if tipoop = 2 then

end if;
if tipoop = 3 then
result := num1 * num2;
end if;
if tipoop = 4 then
result := num1 / num2;
end if;
return result;
end fun;

end if;
if tipoop = 3 then
result := rnum1 * rnum2;

if tipoop = 4 then

end if;
return result;
end fun;
